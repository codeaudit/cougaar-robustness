//PLAYS FOR "NodeFOO"  
// Make a copy of this (changing the name) for EVERY Managed Node
 
 
// Combine local conditions to set the DefenseApplicable condition for the Coordinator
 
(PlannedDisconnect.UnscheduledDisconnect.Node.NodeFOO == TRUE)
            : ApplicableNodeFOO = TRUE;
 
(PlannedDisconnect.ScheduledDisconnect.Node.NodeFOO == TRUE)
            : ApplicableNodeFOO = TRUE;
 
 
(PlannedDisconnect.ScheduledDisconnect.Node.NodeFOO == FALSE) & 
(PlannedDisconnect.UnscheduledDisconnect.Node.NodeFOO == FALSE)
            : ApplicableNodeFOO = FALSE;
 
 
ApplicableNodeFOO == TRUE
            : PlannedDisconnect.Applicable.Node.NodeFOO = TRUE;
 
ApplicableNodeFOO == FALSE
            : PlannedDisconnect.Applicable.Node.NodeFOO = FALSE;
 
 
 
 
// Coordinator logic says this Defense always gets to do what it wants
 
ApplicableNodeFOO == TRUE 
            : DefenseNodeFOO = ENABLED
            : MonitoringNodeFOO = ENABLED;
 
ApplicableNodeFOO == FALSE 
            : DefenseNodeFOO = DISABLED
            : MonitoringNodeFOO = DISABLED;
 
 
DefenseNodeFOO == ENABLED
            : PlannedDisconnect.Defense.Node.NodeFOO = ENABLED;
 
DefenseNodeFOO == DISABLED
            : PlannedDisconnect.Defense.Node.NodeFOO = DISABLED;
 
MonitoringNodeFOO == ENABLED
            : PlannedDisconnect.Monitoring.Node.NodeFOO = ENABLED;
 
MonitoringNodeFOO == DISABLED
            : PlannedDisconnect.Monitoring.Node.NodeFOO = DISABLED;
 
 
 
 
// Transfer OpModes back to the right Defense instances
 
(MonitoringNodeFOO == ENABLED)  &
(PlannedDisconnect.UnscheduledDisconnect.Node.NodeFOO == TRUE) & 
(PlannedDisconnect.UnscheduledReconnectTime.Node.NodeFOO > 0)
            : [agent.NodeFOO]PlannedDisconnect.NodeMonitoring.Node.NodeFOO = DISABLED
            : PlannedDisconnect.ManagerMonitoring.Node.NodeFOO = ENABLED;
 
(MonitoringNodeFOO == ENABLED) &
(PlannedDisconnect.ScheduledDisconnect.Node.NodeFOO == TRUE) & 
(PlannedDisconnect.ScheduledReconnectTime.Node.NodeFOO > 0)
            : [agent.NodeFOO]PlannedDisconnect.NodeMonitoring.Node.NodeFOO = DISABLED
            : PlannedDisconnect.ManagerMonitoring.Node.NodeFOO = ENABLED;
 
(MonitoringNodeFOO == DISABLED)
            : PlannedDisconnect.ManagerMonitoring.Node.NodeFOO = DISABLED
            : [agent.NodeFOO]PlannedDisconnect.NodeMonitoring.Node.NodeFOO = DISABLED;
 
(PlannedDisconnect.ScheduledReconnectTime.Node.NodeFOO == 0) &
(PlannedDisconnect.UnscheduledReconnectTime.Node.NodeFOO == 0)
            : PlannedDisconnect.ManagerMonitoring.NodeFOO = DISABLED
            : [agent.NodeFOO]PlannedDisconnect.NodeMonitoring.NodeFOO = DISABLED;
 
(DefenseNodeFOO == ENABLED)
            : PlannedDisconnect.ManagerDefense.Node.NodeFOO = DISABLED
            : [agent.NodeFOO]PlannedDisconnect.NodeDefense.Node.NodeFOO = ENABLED;
 
(DefenseNodeFOO == DISABLED)
            : PlannedDisconnect.ManagerDefense.Node.NodeFOO = DISABLED
            : [agent.NodeFOO]PlannedDisconnect.NodeDefense.Node.NodeFOO = DISABLED;
